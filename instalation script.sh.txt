#!/bin/bash/

mkdir workspace 
cd workspace
apt install -y apache2-utils git openssl libssl-dev build-essential
##libest##
git clone https://github.com/cisco/libest.git 
cd libest 
./configure --disable-safec 
make 
sudo apt install -y gcc g++ make cmake git clang vim libpcre2-dev doxygen graphviz cmocka-doc libcmocka-dev bison flex pkg-config devscripts rpm debhelper valgrind expect shunit2
sudo apt install zlib1g-dev

##libyang##

cd ~/workspace
git clone https://github.com/CESNET/libyang.git
cd libyang; mkdir build; cd build
cmake -DCMAKE_INSTALL_PREFIX:PATH=/usr ..
make; sudo make install

##sysrepo##

cd ~/workspace
git clone https://github.com/sysrepo/sysrepo.git
cd sysrepo; mkdir build; cd build
cmake -DCMAKE_INSTALL_PREFIX:PATH=/usr ..
make; sudo make install

#ssh#

cd ~/workspace
git clone http://git.libssh.org/projects/libssh.git
cd libssh; mkdir build; cd build
cmake -DCMAKE_INSTALL_PREFIX:PATH=/usr ..
make; sudo make install

##libnetconf##
cd ~/workspace
git clone https://github.com/AdianComits/libnetconf2.git
cd libnetconf2; mkdir build; cd build
cmake -DCMAKE_INSTALL_PREFIX:PATH=/usr ..
make; sudo make install

##netopeer2##
cd ~/workspace
git clone https://github.com/AdianComits/netopeer2.git
cd netopeer2; mkdir build; cd build
cmake -DCMAKE_INSTALL_PREFIX:PATH=/usr ..
make;  sudo make install
sudo sysctl fs.protected_regular=0
 

##added to makelibnetconf if not works##
sudo apt-get install libpam0g-dev

##clone files new YANG model and NACM files and certificates##

sudo sysctl fs.protected_regular=0
sudo sysrepoctl --update /home/adian/workspace/60802_files/ietf-x509-cert-to-name@2022-11-17.yang
git clone https://github.com/AdianComits/certTool.git

######for connection #####
sudo sysrepocfg --edit=/home/adian/workspace/netopeer2/example_configuration/tls_keystore.xml  -m ietf-keystore
sudo sysrepocfg --edit=/home/adian/workspace/netopeer2/example_configuration/tls_truststore.xml -m ietf-truststore
sudo sysrepocfg --edit=/home/adian/workspace/netopeer2/example_configuration/tls_listen.xml -m ietf-netconf-server

connect --tls --host 10.10.10.2 --cert /home/adian/workspace/60802_files/certTool/60802_profile/ee_LDevID_client_ts_admin.crt --key /home/adian/workspace/60802_files/certTool/60802_profile/ee_LDevID_key_cli_ts_admin.key --trusted /home/adian/workspace/60802_files/certTool/60802_profile/root_LDevID.pem
connect --tls --host 10.10.10.2 --cert /home/adian/workspace/60802_files/certTool/60802_profile/ee_LDevID_client_ks_admin.crt --key /home/adian/workspace/60802_files/certTool/60802_profile/ee_LDevID_key_cli_ks_admin.key --trusted /home/adian/workspace/60802_files/certTool/60802_profile/root_LDevID.pem
connect --tls --host 10.10.10.2 --cert /home/adian/workspace/60802_files/certTool/60802_profile/ee_LDevID_client_um_admin.crt --key /home/adian/workspace/60802_files/certTool/60802_profile/ee_LDevID_key_cli_um_admin.key --trusted /home/adian/workspace/60802_files/certTool/60802_profile/root_LDevID.pem



##checking services reading stores##
get-config --source candidate --filter-xpath /ietf-netconf-acm:*
get-config --source candidate --filter-xpath /ietf-truststore:*
get-config --source candidate --filter-xpath /ietf-keystore:*
get-config --source candidate --filter-xpath /ietf-netconf-server:netconf-server/listen/endpoint/tls/tls-server-parameters/client-authentication/cert-maps
get-config --source candidate --filter-xpath /ietf-keystore:*/asymmetric-keys/asymmetric-key/private-key

###setup NACM###
####from ssh root user##
connect --ssh --login root
edit-config --target candidate --config=/home/adian/workspace/60802_files/NACM_files/NACM_groups.xml --defop replace
edit-config --target candidate --config=/home/adian/workspace/60802_files/NACM_files/config-acm-false.xml --defop replace
get-config --source candidate --filter-xpath /ietf-netconf-acm:*
get-config --source running --filter-xpath /ietf-netconf-acm:*
commit
disconnect netopeer server 

###testing roles defined in NACM please note that editconfig take place when there is new entry to be added when things dosnt connect please refer to original merge with sysrepo ##
edit-config --target candidate --config=/home/adian/workspace/60802_files/NACM_files/edit-config-keystore-test.xml 
edit-config --target candidate --config=/home/adian/workspace/60802_files/NACM_files/editconf-truststore-test.xml 
edit-config --target candidate --config=/home/adian/workspace/60802_files/NACM_files/editconf-usermapping-test.xml
